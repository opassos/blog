Repetir n-vezes 

historico_rmse = []
## Hiperpar√¢metros
n_epoch = 10000
lr = 0.05
## Inicializa√ß√£o
a = torch.randn(1).requires_grad_()
b = torch.randn(1).requires_grad_()
## Otimiza√ß√£o
for epoch in range(n_epoch):
    ## Predi√ß√£o
    y_pred = a * x + b
    ## Calcula perda (loss)
    erro = y - y_pred
    rmse = (erro**2).mean()**0.5
    if epoch % 1000 == 0:
        print(f'RMSE: {rmse.item()}')
    historico_rmse.append(rmse.item())
    ## Calcula gradiente
    rmse.backward()
    ## Step
    a.data = a.data - lr * a.grad
    b.data = b.data - lr * b.grad
    a.grad.zero_()
    b.grad.zero_()
